[bml.mod id="bml.state.reload.smoke" name="BML AngelScript State Reload Smoke" version="2.0.0" author="BML" bml="0.3.0" reload="auto" description="Smoke test for script hot reload state migration RestoreState failure."]
class BMLStateReloadSmokeMod {
  void MigrateState(const string &in version, BML::StateBag@ state) {
    if (state !is null) {
      state.SetString("fromVersion", version);
    }
  }

  void RestoreState(BML::StateBag@ state) {
    throw("intentional state reload restore failure smoke");
  }

  void OnLoad(const BML::ModContext &in ctx) {
    BML::Logger@ logger = ctx.BorrowLogger();
    if (logger !is null) {
      logger.Info("BML state reload smoke restore-fail candidate should not load");
    }
  }
}
